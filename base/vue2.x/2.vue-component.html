# 1.2 functional

  • 组件只用于展示数据，所有动态数据都从父组件传递进来（只有 props），内部没有逻辑交互（无 methods 方法、也没有 mounted 等任何生命周期处理函数），没有状态修改(无 data)，推荐使用函数式组件来提升 vue 的性能
<template functional>
  <ul>
    <li
      v-for="(item, index) in props.items"
      :key="index"
      @click="props.itemClick(item)"
    >
      {{ item }}
    </li>
  </ul>
</template>

# 1.3 render

<script>
export default {
  data() {
    return {
      show: false,
    }
  },
  render: (h) => {
    let childNode
    if (this.show) {
      childNode = h("p", "内容 1")
    } else {
      childNode = h("p", "内容 2")
    }
    return h(
      "div",
      {
        attrs: {
          id: "main",
        },
        class: {
          container: true,
        },
        style: {
          color: "red",
        },
      },
      [childNode]
    )
  },
}
</script>

# 3.extend

  • 场景:vue 组件中有些需要将一些元素挂载到元素上,这个时候 extend 就起到作用了
var Profile = Vue.extend({
  template: "<p>{{extendData}}</br>实例传入的数据为:{{propsExtend}}</p>", //template对应的标签最外层必须只有一个标签
  data: function () {
    return {
      extendData: "这是extend扩展的数据",
    }
  },
  props: ["propsExtend"],
})

// 创建的构造器可以挂载到元素上,也可以通过 components 或 Vue.component()注册使用

// 挂载到一个元素上。可以通过propsData传参.

new Profile({ propsData: { propsExtend: "我是实例传入的数据" } }).$mount(
  "#app-extend"
)

import child from "./child"
export default {
  name: "parent",
  component: {
    child,
  },
  mounted() {
    const h = this.$createElement
    let vNode = h(child, {
      props: {},
      methods: {},
    })
    this.$confirm("提示", {

      showCancelButton: true,
      confirmButtonText: "确定",

# 5.patch

<template>
  <div>
    <div id="parent"></div>
  </div>
</template>
<script>
import child from "./child"
export default {
  name: "parent",
  component: {
    child,
  },
  mounted() {
    const h = this.$createElement
    let vNode = h(child, {
      props: {},
      methods: {},
    })
    const dom = document.getElementById("parent")
    this.__patch__(dom, vNode)
  },
}
</script>
